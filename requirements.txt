add an .env file in the root with the following:

# spotify
SPOTIFY_CLIENT_ID=your_spotify_client_id
SPOTIFY_CLIENT_SECRET=your_spotify_client_id
SPOTIFY_REDIRECT_URI=http://127.0.0.1:8080/auth/callback

# gradient
GRADIENT_API_URL=https://api.gradient.run/v1 // replace with your default gradient
DIGITALOCEAN_INFERENCE_KEY=your_digitalocean_inference_key

# app
SESSION_SECRET=your_session_secret // can be any random string
